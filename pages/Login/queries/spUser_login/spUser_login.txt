update users
	set last_login = current_timestamp()
	where email = {{appsmith.user.email}}
	;
	select a.id user_id, b.id acct_id, b.name acct_name, b.default_location procLocation, a.role roleID
	from users a
	join accounts b
	on a.default_account_id = b.id
	where email = {{appsmith.user.email}}
	;
