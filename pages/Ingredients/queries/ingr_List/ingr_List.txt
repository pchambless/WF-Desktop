select
  a.ingr_type	
,	a.ingr_name name
, a.ingr_code code
, a.ingr_desc description
, a.gms_per_oz
,	count(ingr_btch_id) batches
, sum(a.prd_btch_cnt) prod_batches
,	sum(a.purch_qty)  tot_qty
,	cast(min(a.unit_rate) as dec(5,2)) 	min_rate
,	cast(max(a.unit_rate) as dec(5,2)) max_rate
,	cast(avg((ifnull(a.unit_price,0) / ifnull(a.unit_qty,1))) as dec(5,2)) avg_rate
,	a.purch_meas      units
,	sum(a.purch_total_amt) total_cost
, a.ingr_type_id
,  a.ingr_id
,  a.acct_id		
from    	whatsfresh.v_ingr_btch_dtl a
where 	a.ingr_type_id = {{sel_Type.selectedOptionValue	}}	
and     ingr_active = 'Y'
group BY a.ingr_type, a.ingr_name, ingr_code, ingr_desc, a.purch_meas, a.ingr_id, a.acct_id
order by a.ingr_type, a.ingr_name
